git commit --amend 		Отменяет последний коммит и создает новый.
git rm --cached FileName 	Отменяет индексацию файла(git add) после последнего коммита
git rm 	FileName		Удаляет файл только из репозитория.
git remote add 			Добавляем удаленный репозиторий.
git tag	TagName			Для того чтобы создавать метки, для последнего коммита. (По сути своебразные ID)
git tag -a TagName -m Comment	Для того, чтобы создать метку с комментарием
git tag -a TagName -m Comment CheckSumOfCommit	Для того, чтобы создать метку с комментарием для конкретного коммита (указывая его проверочную-сумму)
git mv OldFileName NewFileName	Правильный способ изменения имени файла, не убирая его из репозитория
